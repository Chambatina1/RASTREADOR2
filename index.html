// index.js - Servidor Chambatina // Este archivo va en la raíz de tu proyecto, junto a la carpeta /public const express = require('express'); const axios = require('axios'); const cheerio = require('cheerio'); const path = require('path'); const app = express(); const PORT = process.env.PORT || 3000; // Middlewares app.use(express.json()); app.use(express.static('public')); // Aquí sirve tu index.html, imágenes, etc. // ==================== ENDPOINT PRINCIPAL: RASTREO POR CARNET ==================== // El frontend llama a: /api/rastreo/carnet/12345678901 app.get('/api/rastreo/carnet/:carnet', async (req, res) => { const carnet = req.params.carnet.trim(); if (!carnet) { return res.status(400).json({ error: 'Debes ingresar un número de carnet' }); } try { // 1. Obtener la página del formulario de Kanguro (para posibles tokens) const baseUrl = 'https://www.solvedc.com/tracking/kanguro/'; const formRes = await axios.get(baseUrl, { headers: { 'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36' }, timeout: 10000 }); const $ = cheerio.load(formRes.data); // Buscar token CSRF (si existe) let csrfToken = ''; const tokenSelectors = [ 'input[name="csrf_token"]', 'input[name="_token"]', 'input[name="authenticity_token"]' ]; for (const selector of tokenSelectors) { const token = $(selector).val(); if (token) { csrfToken = token; break; } } // Obtener la URL de envío (action del formulario) let actionUrl = baseUrl; const formAction = $('form').attr('action'); if (formAction) { actionUrl = formAction.startsWith('http') ? formAction : new URL(formAction, baseUrl).href; } // 2. Preparar los datos del formulario const formData = new URLSearchParams(); // IMPORTANTE: El sitio puede esperar 'carnet', 'cedula' o 'identificacion'. // Si no funciona, prueba cambiando el nombre del campo. formData.append('carnet', carnet); if (csrfToken) formData.append('csrf_token', csrfToken); // 3. Enviar la petición POST con el carnet const consultaRes = await axios.post(actionUrl, formData, { headers: { 'Content-Type': 'application/x-www-form-urlencoded', 'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36', 'Referer': baseUrl }, timeout: 15000 }); // 4. Parsear el HTML de la respuesta const $results = cheerio.load(consultaRes.data); const paquetes = []; // --- AQUÍ DEBES AJUSTAR LOS SELECTORES SEGÚN EL HTML REAL DE KANGURO --- // Inspecciona la página con F12 y reemplaza estas clases por las reales. // Por ahora usamos selectores genéricos que capturan tablas o tarjetas comunes. $results('.result-item, .tracking-card, .table tbody tr, .tracking-result').each((i, elem) => { const $elem = $results(elem); const cpk = $elem.find('.guia, .tracking-number, .codigo, td:first-child').text().trim(); const estado = $elem.find('.estado, .status, td:nth-child(2)').text().trim(); const fecha = $elem.find('.fecha, .date, td:nth-child(3)').text().trim(); const descripcion = $elem.find('.producto, .descripcion, td:nth-child(4)').text().trim(); const observaciones = $elem.find('.observaciones, .nota, td:nth-child(5)').text().trim(); if (cpk) { paquetes.push({ cpk, estado, fecha, descripcion, observaciones, timeline: [estado] // Podrías extraer más historial si existe }); } }); // Fallback: si no se encontró con los selectores, buscar cualquier CPK en el texto if (paquetes.length === 0) { const bodyText = $results.body.text(); const cpkMatch = bodyText.match(/CPK-[A-Z0-9]+/); if (cpkMatch) { paquetes.push({ cpk: cpkMatch[0], estado: 'Ver detalles en el sistema', fecha: '', descripcion: bodyText.substring(0, 200).replace(/\s+/g, ' '), observaciones: '', timeline: [] }); } } if (paquetes.length === 0) { return res.status(404).json({ error: 'No se encontraron paquetes para ese carnet' }); } // 5. Devolver los datos al frontend de Chambatina res.json(paquetes); } catch (error) { console.error('Error en proxy:', error.message); res.status(500).json({ error: 'El servicio de rastreo no está disponible. Intenta más tarde.' }); } }); // ==================== ENDPOINT DE CHAT (BÁSICO) ==================== app.post('/api/chat', (req, res) => { const { message } = req.body; let reply = 'Hola, soy Chambita. Puedo ayudarte con el rastreo (usa tu carnet) o con información sobre productos solares.'; if (message && message.toLowerCase().includes('precio')) { reply = 'Para precios personalizados, escríbenos por WhatsApp.'; } else if (message && message.toLowerCase().includes('solar')) { reply = 'Ofrecemos kits solares, inversores y baterías. ¿Necesitas asesoría? Contáctanos.'; } res.json({ reply }); }); // ==================== INICIAR SERVIDOR ==================== app.listen(PORT, () => { console.log(`✅ Servidor Chambatina corriendo en http://localhost:${PORT}`); console.log(`📦 Ejemplo de consulta: http://localhost:${PORT}/api/rastreo/carnet/12345678901`); });
