// index.js - Servidor Chambatina (proxy para rastreo por carnet) const express = require('express'); const axios = require('axios'); const cheerio = require('cheerio'); const path = require('path'); const app = express(); const PORT = process.env.PORT || 3000; // Middlewares app.use(express.json()); app.use(express.static('public')); // Aquí va tu frontend (HTML, CSS, imágenes) // ==================== RASTREO POR CARNET ==================== // Endpoint que usa el frontend de Chambatina app.get('/api/rastreo/carnet/:carnet', async (req, res) => { const carnet = req.params.carnet.trim(); if (!carnet) { return res.status(400).json({ error: 'Debes ingresar un número de carnet' }); } try { // 1. Obtener la página del formulario (para posibles tokens CSRF) const baseUrl = 'https://www.solvedc.com/tracking/kanguro/'; const formRes = await axios.get(baseUrl, { headers: { 'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36' }, timeout: 10000 }); const $ = cheerio.load(formRes.data); // Buscar token CSRF (si existe) let csrfToken = ''; const tokenSelectors = ['input[name="csrf_token"]', 'input[name="_token"]', 'input[name="authenticity_token"]']; for (const sel of tokenSelectors) { const val = $(sel).val(); if (val) { csrfToken = val; break; } } // Determinar URL de envío (action del formulario) let actionUrl = baseUrl; const formAction = $('form').attr('action'); if (formAction) actionUrl = formAction.startsWith('http') ? formAction : new URL(formAction, baseUrl).href; // 2. Preparar datos del formulario const formData = new URLSearchParams(); // ⚠️ CAMBIA EL NOMBRE DEL CAMPO si el sitio usa 'cedula' o 'identificacion' formData.append('carnet', carnet); if (csrfToken) formData.append('csrf_token', csrfToken); // 3. Enviar petición POST const consultaRes = await axios.post(actionUrl, formData, { headers: { 'Content-Type': 'application/x-www-form-urlencoded', 'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36', 'Referer': baseUrl }, timeout: 15000 }); // 4. Parsear respuesta HTML const $results = cheerio.load(consultaRes.data); const paquetes = []; // ⚠️ AQUÍ DEBES CAMBIAR LOS SELECTORES SEGÚN EL HTML REAL DE KANGURO // Ejemplo genérico: busca contenedores con clase 'result-item' o 'tracking-card' $results('.result-item, .tracking-card, .table tbody tr').each((i, elem) => { const $elem = $results(elem); const cpk = $elem.find('.guia, .tracking-number, td:first-child').text().trim(); const estado = $elem.find('.estado, .status, td:nth-child(2)').text().trim(); const fecha = $elem.find('.fecha, .date, td:nth-child(3)').text().trim(); const descripcion = $elem.find('.producto, .descripcion, td:nth-child(4)').text().trim(); const observaciones = $elem.find('.observaciones, .nota, td:nth-child(5)').text().trim(); if (cpk) { paquetes.push({ cpk, estado, fecha, descripcion, observaciones, timeline: [estado] }); } }); // Fallback: si no encuentra con los selectores, busca cualquier CPK en el texto if (paquetes.length === 0) { const bodyText = $results.body.text(); const cpkMatch = bodyText.match(/CPK-[A-Z0-9]+/); if (cpkMatch) { paquetes.push({ cpk: cpkMatch[0], estado: 'Ver detalles en el sistema', fecha: '', descripcion: bodyText.substring(0, 200), observaciones: '', timeline: [] }); } } if (paquetes.length === 0) { return res.status(404).json({ error: 'No se encontraron paquetes para ese carnet' }); } res.json(paquetes); } catch (error) { console.error('Error en proxy:', error.message); res.status(500).json({ error: 'El servicio de rastreo externo no está disponible. Intenta más tarde.' }); } }); // ==================== CHAT BÁSICO (opcional) ==================== app.post('/api/chat', (req, res) => { const { message } = req.body; let reply = 'Hola, soy Chambita. Usa el rastreador con tu carnet o pregúntame sobre productos solares.'; if (message && message.toLowerCase().includes('precio')) { reply = 'Contáctanos por WhatsApp para cotizaciones personalizadas.'; } res.json({ reply }); }); // ==================== INICIAR SERVIDOR ==================== app.listen(PORT, () => { console.log(`✅ Chambatina server en http://localhost:${PORT}`); console.log(`📦 Ejemplo: http://localhost:${PORT}/api/rastreo/carnet/12345678901`); });
